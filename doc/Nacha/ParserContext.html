def initialize(opts = {})
  @file_name = opts[:file_name] || ''
  @line_number = opts[:line_number] || 0
  @line_length = opts[:line_length] || 0
  @current_line_errors = []
  @parser_started_at = Time.now.utc
  @parser_ended_at = nil
  @previous_record = nil
  @validated = false
end
