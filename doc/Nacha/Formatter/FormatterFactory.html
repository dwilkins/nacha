def self.get(format, records, options = {})
  case format
  when :json
    JsonFormatter.new(records, options)
  when :html
    HtmlFormatter.new(records, options)
  when :markdown
    MarkdownFormatter.new(records, options)
  else
    raise ArgumentError, "Unknown format: #{format}"
  end
end
